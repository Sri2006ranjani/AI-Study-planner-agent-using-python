# ===== FUNCTION TO GENERATE STUDY PLAN =====
def generate_study_plan(subjects, difficulties, total_hours):
    # Weight for each difficulty
    weights = {
        "easy": 1,
        "medium": 2,
        "hard": 3
    }

    # Calculate total weight
    total_weight = 0
    for d in difficulties:
        total_weight += weights[d]

    # Allocate time per subject
    study_plan = {}
    for subject, difficulty in zip(subjects, difficulties):
        allocated_time = (weights[difficulty] / total_weight) * total_hours
        study_plan[subject] = round(allocated_time, 2)

    return study_plan


# ===== MAIN PROGRAM =====
print("üß† AI STUDY PLANNER \n")

subjects_input = input("Enter subjects (comma separated): ")
difficulty_input = input("Enter difficulty for each subject (easy/medium/hard): ")
total_hours = float(input("Enter total study hours per day: "))

subjects = [s.strip() for s in subjects_input.split(",")]
difficulties = [d.strip().lower() for d in difficulty_input.split(",")]

# Validation
if len(subjects) != len(difficulties):
    print("‚ùå Number of subjects and difficulties must match.")
    exit()

# Generate study plan
plan = generate_study_plan(subjects, difficulties, total_hours)

# Display study plan
print("\nüìã Your Daily Study Plan:")
for subject, hours in plan.items():
    print(f"  - {subject}: {hours} hours")

print("\n‚úÖ Study plan generated successfully!")
